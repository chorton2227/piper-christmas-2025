// Utility sleep function
function sleep(ms) {

// Staggered card dealing
function displayHandAnimated(hand, containerId, delay = 200) {
    hand.forEach((card, index) => {
        setTimeout(() => {
            let cardElement = createCardElement(card);
            cardElement.classList.add('dealing');
            container.appendChild(cardElement);
            setTimeout(() => cardElement.classList.remove('dealing'), 500);
        }, index * delay);
    });
}

    let rankCounts = {};

    }

// Finding multiples (pattern detection)
function findMultiples(rankCounts) {
    let pairs = [], threeOfAKind = null, fourOfAKind = null;
    for (let rank in rankCounts) {
        if (rankCounts[rank] === 4) fourOfAKind = rank;
        else if (rankCounts[rank] === 3) threeOfAKind = rank;
